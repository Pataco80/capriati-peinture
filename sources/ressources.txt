gatsby-plugin-minify



graphql queries works gallery

query worksGalery {
  work1Images: allFile(filter: {relativePath: {regex: "/work1/"}}) {
    ...FileConnectionFragment
  }
  work2Images: allFile(filter: {relativePath: {regex: "/work2/"}}) {
    ...FileConnectionFragment
  }
}

fragment FileConnectionFragment on FileConnection {
  nodes {
    relativePath
    childImageSharp {
      fluid {
        src
        originalName
      }
    }
  }
  edges {
    node {
      relativeDirectory
    }
  }
}
